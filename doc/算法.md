### 二分查找法-比较次数少，查找速度快 但是排序的数组必须是已经排序的
    public static int binarySearch(int a[], int key) {
      int left = 0;
      int right = a.length - 1;
      int mid;
      while (left <= right) {
        mid = (left + right) / 2;
        if (key == a[mid])
          return mid;
        if (key < a[mid])
          right = mid - 1;// right = mid; 可能死循环
        else
          left = mid + 1;// left = mid; 可能死循环
      }
      return -1;
    }
